library(magrittr)

moeda <- function(p){
    x <- runif(1) %>% "<"(p) %>% ifelse(1,0)
}

moedas <- function(p,n){
    sapply(1:n, function(x)moeda(p))
}

proporcao <- function(resultados){
    sum(resultados)/length(resultados)
}

simulacao <- function(k,n,p){
    sapply(1:k, function(x)proporcao(moedas(p,n)))
}
